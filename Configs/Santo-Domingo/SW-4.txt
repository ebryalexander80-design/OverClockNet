enable
configure terminal
hostname SW4
no ip domain-lookup
ip domain-name overclocknet.com.do
enable secret cisco123
service password-encryption

username admin privilege 15 secret Admin-Overclock2026

crypto key generate rsa modulus 2048
ip ssh version 2
ip ssh time-out 60
ip ssh authentication-retries 3

line console 0
 login local
 exec-timeout 5 0
 logging synchronous
exit

line vty 0 4
 login local
 transport input ssh
 exec-timeout 5 0
exit
