----------------------------------------------------------------
ISP - CONFIGURACIÓN
----------------------------------------------------------------

hostname ISP
no ip domain-lookup
enable secret cisco123
service password-encryption
banner motd # ISP - ACCESO RESTRINGIDO #

========================
          SSH
========================
username admin privilege 15 secret Isp-access2026
crypto key generate rsa modulus 2048
ip ssh version 2

line console 0
 login local
 exec-timeout 5 0
 logging synchronous
exit

line vty 0 4
 login local
 transport input ssh
 exec-timeout 5 0
exit
